343 void process(
345 Context & ctx ) override
346 {
347 m_isDefined = true;
348
349 switch( m_opt )
350 {
351 case ValueOptions::ManyValues :
352 {
353 eatValues( ctx, m_values,
354 String( SL( "Command \"" ) ) +
355 name() + SL( "\" requires value that wasn't presented." ),
356 cmdLine() );
357 }
358 break;
359
360 case ValueOptions::OneValue :
361 {
362 m_values.push_back( eatOneValue( ctx,
363 String( SL( "Command \"" ) ) + name() +
364 SL( "\" requires value that wasn't presented." ),
365 cmdLine() ) );
366 }
367 break;
368
369 default :
370 break;

380 void checkCorrectnessBeforeParsing(
382 StringList & flags,
384 StringList & names ) const override
385 {
386 if( details::isCorrectName( name() ) )
387 {
388 auto it = std::find( names.begin(), names.end(), name() );
389
390 if( it != names.end() )
391 throw BaseException( String( SL( "Redefinition of command "
392 "with name \"" ) ) + name() + SL( "\"." ) );
393 else
394 names.push_back( name() );
395 }
396 else
397 throw BaseException( String( SL( "Disallowed name \"" ) ) +
398 name() + SL( "\" for the command." ) );
399
400 StringList ftmp = flags;
401 StringList ntmp = names;
402
403 GroupIface::checkCorrectnessBeforeParsing( ftmp, ntmp );
404 }
