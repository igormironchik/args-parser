582 inline void
583 CmdLine::parse()
584 {
585  clear();
586 
587  checkCorrectnessBeforeParsing();
588 
589  while( !m_context.atEnd() )
590  {
591  String word = *m_context.next();
592 
593  const String::size_type eqIt = word.find( '=' );
594 
595  if( eqIt != String::npos )
596  {
597  const String value = word.substr( eqIt + 1 );
598 
599  if( !value.empty() )
600  m_context.prepend( value );
601 
602  word = word.substr( 0, eqIt );
603  }
604 
605  if( details::isArgument( word ) )
606  {
607  auto * arg = findArgument( word );
608 
609  if( arg )
610  arg->process( m_context );
611  else
612  printInfoAboutUnknownArgument( word );
613  }
614  else if( details::isFlag( word ) )
615  {
616  for( String::size_type i = 1, length = word.length(); i < length; ++i )
617  {
618  const String flag = String( SL( "-" ) ) +
619 
620 #ifdef ARGS_QSTRING_BUILD
621  String( word[ i ] );
622 #else
623  String( 1, word[ i ] );
624 #endif
625 
626  auto * arg = findArgument( flag );
627 
628  if( !arg )
629  throw BaseException( String( SL( "Unknown argument \"" ) ) +
630  flag + SL( "\"." ) );
631 
632  if( i < length - 1 && arg->isWithValue() )
633  throw BaseException( String( SL( "Only last argument in "
634  "flags combo can be with value. Flags combo is \"" ) ) +
635  word + SL( "\"." ) );
636  else
637  arg->process( m_context );
638  }
639  }
640  // Command?
641  else
642  {
643  auto * tmp = findArgument( word );
644 
645  auto check = [this, &tmp, &word] ()
646  {
647  const auto & args = m_args;
648 
649  const auto it = std::find_if( args.cbegin(),
650  args.cend(), [&tmp] ( const auto & arg ) -> bool
651  { return ( arg.get() == tmp ); } );
652 
653  if( m_command && it != args.cend() && !m_command->findChild( word ) )
654  throw BaseException( String( SL(
655  "Only one command from one level can be specified. "
656  "But you defined \"" ) ) +
657  word + SL( "\" and \"" ) + m_command->name() + SL( "\"." ) );
658  };
659 
660  if( tmp )
661  {
662  if( tmp->type() == ArgType::Command )
663  {
664  if( !m_prevCommand.empty() )
665  {
666  for( const auto & prev : m_prevCommand )
667  {
668  const auto & args = prev->children();
669 
670  const auto it = std::find_if( args.cbegin(),
671  args.cend(), [&tmp] ( const auto & arg ) -> bool
672  { return ( arg.get() == tmp ); } );
673 
674  if( it != args.cend() && !m_currCommand->findChild( word ) )
675  throw BaseException( String( SL(
676  "Only one command from one level can be specified. "
677  "But you defined \"" ) ) +
678  word + SL( "\" and \"" ) + prev->name() + SL( "\"." ) );
679  }
680 
681  check();
682  }
683  else
684  check();
685 
686  if( !m_command )
687  {
688  m_command = static_cast< Command* > ( tmp );
689 
690  m_currCommand = m_command;
691 
692  m_prevCommand.push_back( m_command );

699
