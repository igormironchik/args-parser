500inline void
501CmdLine::parse()
502{
503 clear();
504
505 checkCorrectnessBeforeParsing();
506
507 while( !m_context.atEnd() )
508 {
509 String word = *m_context.next();
510
511 const String::size_type eqIt = word.find( '=' );
512
513 if( eqIt != String::npos )
514 {
515 const String value = word.substr( eqIt + 1 );
516
517 if( !value.empty() )
518 m_context.prepend( value );
519
520 word = word.substr( 0, eqIt );
521 }
522
523 if( details::isArgument( word ) )
524 {
525 auto * arg = findArgument( word );
526
527 if( arg )
528 arg->process( m_context );
529 else
530 printInfoAboutUnknownArgument( word );
531 }
532 else if( details::isFlag( word ) )
533 {
534 for( String::size_type i = 1, length = word.length(); i < length; ++i )
535 {
536 const String flag = String( SL( "-" ) ) +
537
538#ifdef ARGS_QSTRING_BUILD
539 String( word[ i ] );
540#else
541 String( 1, word[ i ] );
542#endif
543
544 auto * arg = findArgument( flag );
545
546 if( !arg )
547 throw BaseException( String( SL( "Unknown argument \"" ) ) +
548 flag + SL( "\"." ) );
549
550 if( i < length - 1 && arg->isWithValue() )
551 throw BaseException( String( SL( "Only last argument in "
552 "flags combo can be with value. Flags combo is \"" ) ) +
553 word + SL( "\"." ) );
554 else
555 arg->process( m_context );
556 }
557 }
558 // Command?
559 else
560 {
561 auto * tmp = findArgument( word );
562
563 auto check = [this, &tmp, &word] ()
564 {
565 const auto & args = m_args;
566
567 const auto it = std::find_if( args.cbegin(),
568 args.cend(), [&tmp] ( const auto & arg ) -> bool
569 { return ( arg.get() == tmp ); } );
570
571 if( m_command && it != args.cend() && !m_command->findChild( word ) )
572 throw BaseException( String( SL(
573 "Only one command from one level can be specified. "
574 "But you defined \"" ) ) +
575 word + SL( "\" and \"" ) + m_command->name() + SL( "\"." ) );
576 };
577
578 if( tmp )
579 {
580 if( tmp->type() == ArgType::Command )
581 {
582 if( !m_prevCommand.empty() )
583 {
584 for( const auto & prev : m_prevCommand )
585 {
586 const auto & args = prev->children();
587
588 const auto it = std::find_if( args.cbegin(),
589 args.cend(), [&tmp] ( const auto & arg ) -> bool
590 { return ( arg.get() == tmp ); } );
591
592 if( it != args.cend() && !m_currCommand->findChild( word ) )
593 throw BaseException( String( SL(
594 "Only one command from one level can be specified. "
595 "But you defined \"" ) ) +
596 word + SL( "\" and \"" ) + prev->name() + SL( "\"." ) );
597 }
598
599 check();
600 }
601 else
602 check();
603
604 if( !m_command )
605 {
606 m_command = static_cast< Command* > ( tmp );
607
608 m_currCommand = m_command;
609
610 m_prevCommand.push_back( m_command );
611
612 m_command->process( m_context );
613 }
614 else
615 {
616 auto * cmd = static_cast< Command* > ( tmp );
617
618 m_currCommand->setCurrentSubCommand( cmd );
619
620 m_currCommand = cmd;
621
622 m_prevCommand.push_back( m_currCommand );
623
624 m_currCommand->process( m_context );
625 }
626 }
627 else
628 printInfoAboutUnknownArgument( word );
629 }
630 else
631 printInfoAboutUnknownArgument( word );
632 }
633 }
634
635 checkCorrectnessAfterParsing();
636}
